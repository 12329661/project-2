# make function to standardize sex values
def clean_sex(s):
    if pd.isnull(s): # if value is missing its Unknown
        return 'Unknown'
    s = str(s).strip().upper() # remove spaces and make uppercase
    if s == 'M':
        return 'M'
    if s == 'F':
        return 'F'
    return 'Unknown'

# apply to every row in Sex column and store it in new column

def species_group(s):
    if pd.isnull(s):  # missing values = Unknown
        return 'Unknown'
    s = str(s).lower()  # convert to lowercase

    if 'white' in s:
        return 'White Shark'
    if 'tiger' in s:
        return 'Tiger Shark'
    if 'bull' in s:
        return 'Bull Shark'
    if 'nurse' in s:
        return 'Nurse Shark'
    if 'reef' in s:
        return 'Reef Shark'
    if 'hammerhead' in s:
        return 'Hammerhead'
